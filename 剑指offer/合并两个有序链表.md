```cpp
#include <iostream>

using namespace std;


struct ListNode {
	int val;
	struct ListNode *next;
	ListNode(int x) :
			val(x), next(NULL) {
	}
};

class Solution {
public:
    ListNode* Merge(ListNode* pHead1, ListNode* pHead2)
    {
        ListNode* L = (ListNode*)malloc(sizeof(struct ListNode));
        if(!pHead1)
             return pHead2;
        if(!pHead2)
             return pHead1;
        if(pHead1->val < pHead2->val){
            L = pHead1;
            pHead1 = pHead1->next;
        }
        else{
            L = pHead2;
            pHead2 = pHead2->next;
        }
        ListNode* Head = L;   
        while(pHead1&&pHead2){
             if(pHead1->val < pHead2->val){
                 L->next = pHead1;
                 pHead1 = pHead1->next;
                 L = L->next;
             }
             else{
                 L->next = pHead2;
                 pHead2 = pHead2->next;
                 L = L->next;
             }
        }
        if(pHead1){
            L->next = pHead1;
        }
        if(pHead2){
            L->next = pHead2;
        }
        return Head;
    }
};

int main()
{
    cout << "Hello world!" << endl;
    return 0;
}
```
